# Select

import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";
import "@nuralyui/select";
import { HySelect } from "@nuralyui/select/react";
import { useState } from "react";

export const BasicSelectDemo = () => {
  const [selectedOptions, setSelectedOptions] = useState([]);
  
  const options = [
    { value: 'apple', label: '游꼝 Apple' },
    { value: 'banana', label: '游꼛 Banana' },
    { value: 'orange', label: '游꼙 Orange' },
    { value: 'grape', label: '游꼖 Grape' }
  ];
  
  const handleChange = (e) => {
    setSelectedOptions(e.detail.value);
  };
  
  return (
    <div style={{ marginBottom: "20px" }}>
      <HySelect 
        placeholder="Choose an option"
        options={options}
        onChanged={handleChange}
        style={{ width: "200px" }}
      />
      {selectedOptions.length > 0 && (
        <div style={{ marginTop: "10px", color: "#666" }}>
          Selected: {selectedOptions.join(", ")}
        </div>
      )}
    </div>
  );
};

The Select component allows users to choose one or multiple options from a dropdown list. It supports various configurations including single/multi-select, different sizes, and custom styling.

## API Reference

### Properties

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `placeholder` | `string` | `undefined` | Placeholder text when no option is selected |
| `options` | `Array<{value: string, label: string}>` | `[]` | Array of option objects with value and label |
| `selectionMode` | `'single' \| 'multiple'` | `'single'` | Whether single or multiple selections are allowed |
| `defaultSelected` | `string[]` | `[]` | Array of default selected values |
| `disabled` | `boolean` | `false` | Whether the select is disabled |
| `size` | `'small' \| 'medium' \| 'large'` | `'medium'` | The size of the select |
| `status` | `'default' \| 'error' \| 'warning'` | `'default'` | The status/validation state |

### Events

| Event | Description |
|-------|-------------|
| `changed` | Fired when the selection changes, returns `e.detail.value` array |

```jsx
import { HySelect } from "@nuralyui/select/react";
import { useState } from "react";

export default function App() {
  const [selectedValues, setSelectedValues] = useState([]);
  
  const handleChange = (e) => {
    const value = Array.from(e.target.selectedOptions, option => option.value);
    setSelectedValues(value);
  };
  
  return (
    <div>
      <HySelect 
        multiple 
        placeholder="Select multiple options"
        onChange={handleChange}
      >
        <option value="red">游댮 Red</option>
        <option value="green">游릭 Green</option>
        <option value="blue">游댯 Blue</option>
        <option value="yellow">游리 Yellow</option>
        <option value="purple">游릮 Purple</option>
      </HySelect>
      {selectedValues.length > 0 && (
        <div>Selected: {selectedValues.join(", ")}</div>
      )}
    </div>
  );
}
```
