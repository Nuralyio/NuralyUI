# Radio

import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";
import "@nuralyui/radio";
import { HyRadio } from "@nuralyui/radio/react";
import { useState } from "react";

export const BasicRadioDemo = () => {
  const [selectedValue, setSelectedValue] = useState("");
  
  const options = [
    { value: 'option1', label: 'Option 1' },
    { value: 'option2', label: 'Option 2' },
    { value: 'option3', label: 'Option 3' },
    { value: 'option4', label: 'Option 4' }
  ];
  
  const handleChange = (e) => {
    setSelectedValue(e.detail.value);
  };
  
  return (
    <div style={{ marginBottom: "20px" }}>
      <HyRadio
        options={options}
        onChanged={handleChange}
      />
      {selectedValue && (
        <div style={{ marginTop: "10px", color: "#666" }}>
          Selected: {selectedValue}
        </div>
      )}
    </div>
  );
};

The Radio component allows users to select one option from a set of mutually exclusive choices. It supports various styles including traditional radio buttons and button-style selections.

## API Reference

### Properties

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `options` | `RadioButtonOption[]` | `[]` | Array of radio option objects |
| `defaultValue` | `string` | `undefined` | The initially selected value |
| `type` | `'default' \| 'button'` | `'default'` | Display style of radio buttons |
| `direction` | `'vertical' \| 'horizontal'` | `'vertical'` | Layout direction of options |
| `position` | `'left' \| 'right'` | `'left'` | Position of radio button relative to label |

### RadioButtonOption Type

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `value` | `string` | - | The value of the option |
| `label` | `string` | - | The display text for the option |
| `icon` | `string` | `undefined` | Optional icon name |
| `disabled` | `boolean` | `false` | Whether the option is disabled |
| `state` | `'error' \| 'warning'` | `undefined` | Validation state |
| `message` | `string` | `undefined` | Message to display for error/warning states |

### Events

| Event | Description |
|-------|-------------|
| `change` | Fired when the selection changes, returns `e.detail.value` |
