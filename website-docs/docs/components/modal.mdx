# Modal

import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";
import "@nuralyui/modal";
import { HyModal } from "@nuralyui/modal/react";
import { HyButton } from "@nuralyui/button/react";
import { useState } from "react";

The Modal component provides a way to display content in a dialog overlay, capturing user attention and focus. It's perfect for confirmations, forms, or any content that needs to interrupt the user's workflow.

## How to use it

</TabItem>
 <TabItem value="yarn" label="yarn">

## Import

<Tabs>
  <TabItem value="html" label="HTML">

```html
<script type="module">
  import '@nuralyui/modal';
</script>

<hy-modal title="Modal Title" open="false">
  <div slot="content">
    <p>Modal content goes here</p>
  </div>
  <div slot="footer">
    <button>Cancel</button>
    <button>OK</button>
  </div>
</hy-modal>
```

</TabItem>
  <TabItem value="react" label="React">

```jsx
import { HyModal } from '@nuralyui/modal/react';
import { useState } from 'react';

function App() {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <>
      <button onClick={() => setIsOpen(true)}>Open Modal</button>
      <HyModal 
        title="Modal Title" 
        open={isOpen}
        onClose={() => setIsOpen(false)}
      >
        <div slot="content">
          <p>Modal content goes here</p>
        </div>
        <div slot="footer">
          <button onClick={() => setIsOpen(false)}>Cancel</button>
          <button onClick={() => setIsOpen(false)}>OK</button>
        </div>
      </HyModal>
    </>
  );
}
```

## Confirmation Modal

export const ConfirmationModalExample = () => {
  const [isOpen, setIsOpen] = useState(false);

  const handleConfirm = () => {
    alert('Action confirmed!');
    setIsOpen(false);
  };

  return (
    <div>
      <HyButton type="primary" onClick={() => setIsOpen(true)}>Delete Item</HyButton>
      <HyModal 
        title="Confirm Deletion" 
        open={isOpen}
        onClose={() => setIsOpen(false)}
      >
        <div slot="content">
          <p>Are you sure you want to delete this item?</p>
          <p>This action cannot be undone.</p>
        </div>
        <div slot="footer">
          <HyButton type="dashed" onClick={() => setIsOpen(false)}>Cancel</HyButton>
          <HyButton type="primary" style={{backgroundColor: '#ff4d4f'}} onClick={handleConfirm}>
            Delete
          </HyButton>
        </div>
      </HyModal>
    </div>
  );
};

## Form Modal

export const FormModalExample = () => {
  const [isOpen, setIsOpen] = useState(false);
  const [formData, setFormData] = useState({ name: '', email: '' });

  const handleSubmit = () => {
    console.log('Form submitted:', formData);
    setIsOpen(false);
    setFormData({ name: '', email: '' });
  };

  return (
    <div>
      <HyButton type="primary" onClick={() => setIsOpen(true)}>Add User</HyButton>
      <HyModal 
        title="Add New User" 
        open={isOpen}
        onClose={() => setIsOpen(false)}
      >
        <div slot="content">
          <form style={{ display: 'flex', flexDirection: 'column', gap: '16px' }}>
            <div>
              <label style={{ display: 'block', marginBottom: '4px' }}>Name:</label>
              <input 
                type="text" 
                value={formData.name}
                onChange={(e) => setFormData({...formData, name: e.target.value})}
                style={{ width: '100%', padding: '8px', border: '1px solid #d9d9d9', borderRadius: '4px' }}
              />
            </div>
            <div>
              <label style={{ display: 'block', marginBottom: '4px' }}>Email:</label>
              <input 
                type="email" 
                value={formData.email}
                onChange={(e) => setFormData({...formData, email: e.target.value})}
                style={{ width: '100%', padding: '8px', border: '1px solid #d9d9d9', borderRadius: '4px' }}
              />
            </div>
          </form>
        </div>
        <div slot="footer">
          <HyButton type="dashed" onClick={() => setIsOpen(false)}>Cancel</HyButton>
          <HyButton type="primary" onClick={handleSubmit}>Submit</HyButton>
        </div>
      </HyModal>
    </div>
  );
};

## API Reference

### Properties

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `title` | `string` | `undefined` | The title displayed in the modal header |
| `open` | `boolean` | `false` | Controls whether the modal is visible |
| `width` | `string` | `'520px'` | The width of the modal |
| `height` | `string` | `'auto'` | The height of the modal |
| `closable` | `boolean` | `true` | Whether the modal can be closed by clicking the X button |
| `maskClosable` | `boolean` | `true` | Whether the modal can be closed by clicking the backdrop |
| `draggable` | `boolean` | `false` | Whether the modal can be dragged |

### Events

| Event | Description |
|-------|-------------|
| `close` | Fired when the modal is closed |
| `open` | Fired when the modal is opened |

### Slots

| Slot | Description |
|------|-------------|
| `content` | The main content area of the modal |
| `footer` | The footer area, typically containing action buttons |

### CSS Custom Properties

You can customize the modal appearance using CSS custom properties:

```css
hy-modal {
  --modal-background: #ffffff;
  --modal-border-radius: 6px;
  --modal-box-shadow: 0 6px 16px 0 rgba(0, 0, 0, 0.08);
  --modal-header-background: #ffffff;
  --modal-header-border-bottom: 1px solid #f0f0f0;
  --modal-backdrop-background: rgba(0, 0, 0, 0.45);
  --modal-close-color: #00000073;
  --modal-close-hover-color: #000000d9;
}
```
