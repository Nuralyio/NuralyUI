# Toast

import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";
import "@nuralyui/toast";
import { HyToast } from "@nuralyui/toast/react";
import { HyButton } from "@nuralyui/button/react";

The Toast component provides a way to show brief, auto-dismissing messages to users. Perfect for notifications, success messages, warnings, or error alerts that don't require user interaction.

## Basic Example

export const BasicToastExample = () => {
  const [toasts, setToasts] = useState([]);

  const showToast = (message, type = 'info') => {
    const newToast = {
      id: Date.now(),
      text: message,
      type: type
    };
    setToasts(prev => [...prev, newToast]);
    
    // Auto remove after 3 seconds
    setTimeout(() => {
      setToasts(prev => prev.filter(toast => toast.id !== newToast.id));
    }, 3000);
  };

  return (
    <div>
      <HyButton onClick={() => showToast('This is a basic toast message!')}>
        Show Toast
      </HyButton>
      
      <div style={{
        position: 'fixed',
        top: '20px',
        right: '20px',
        zIndex: 1000,
        display: 'flex',
        flexDirection: 'column',
        gap: '8px'
      }}>
        {toasts.map(toast => (
          <div key={toast.id} style={{
            background: '#333',
            color: 'white',
            padding: '12px 16px',
            borderRadius: '6px',
            boxShadow: '0 4px 12px rgba(0, 0, 0, 0.15)',
            animation: 'slideInRight 0.3s ease-out'
          }}>
            {toast.text}
          </div>
        ))}
      </div>
    </div>
  );
};

## Toast Types

export const ToastTypesExample = () => {
  const [toasts, setToasts] = useState([]);

  const showToast = (message, type = 'info') => {
    const newToast = {
      id: Date.now(),
      text: message,
      type: type
    };
    setToasts(prev => [...prev, newToast]);
    
    setTimeout(() => {
      setToasts(prev => prev.filter(toast => toast.id !== newToast.id));
    }, 3000);
  };

  const getToastStyle = (type) => {
    const baseStyle = {
      padding: '12px 16px',
      borderRadius: '6px',
      boxShadow: '0 4px 12px rgba(0, 0, 0, 0.15)',
      color: 'white',
      display: 'flex',
      alignItems: 'center',
      gap: '8px'
    };

    switch (type) {
      case 'success':
        return { ...baseStyle, background: '#52c41a' };
      case 'error':
        return { ...baseStyle, background: '#ff4d4f' };
      case 'warning':
        return { ...baseStyle, background: '#faad14' };
      case 'info':
      default:
        return { ...baseStyle, background: '#1890ff' };
    }
  };

  const getIcon = (type) => {
    switch (type) {
      case 'success': return '✓';
      case 'error': return '✕';
      case 'warning': return '⚠';
      case 'info': return 'ℹ';
      default: return '';
    }
  };

  return (
    <div style={{ display: 'flex', gap: '12px', flexWrap: 'wrap' }}>
      <HyButton 
        type="primary" 
        onClick={() => showToast('Operation successful!', 'success')}
      >
        Success Toast
      </HyButton>
      <HyButton 
        style={{backgroundColor: '#ff4d4f'}}
        onClick={() => showToast('Something went wrong!', 'error')}
      >
        Error Toast
      </HyButton>
      <HyButton 
        style={{backgroundColor: '#faad14'}}
        onClick={() => showToast('Please check your input!', 'warning')}
      >
        Warning Toast
      </HyButton>
      <HyButton 
        onClick={() => showToast('Here is some information', 'info')}
      >
        Info Toast
      </HyButton>
      
      <div style={{
        position: 'fixed',
        top: '20px',
        right: '20px',
        zIndex: 1000,
        display: 'flex',
        flexDirection: 'column',
        gap: '8px'
      }}>
        {toasts.map(toast => (
          <div key={toast.id} style={getToastStyle(toast.type)}>
            <span>{getIcon(toast.type)}</span>
            {toast.text}
          </div>
        ))}
      </div>
    </div>
  );
};

## Toast Positions

export const ToastPositionExample = () => {
  const [toasts, setToasts] = useState([]);
  const [position, setPosition] = useState('top-right');

  const showToast = (message) => {
    const newToast = {
      id: Date.now(),
      text: message,
      position: position
    };
    setToasts(prev => [...prev, newToast]);
    
    setTimeout(() => {
      setToasts(prev => prev.filter(toast => toast.id !== newToast.id));
    }, 3000);
  };

  const getPositionStyle = (pos) => {
    const baseStyle = {
      position: 'fixed',
      zIndex: 1000,
      display: 'flex',
      flexDirection: 'column',
      gap: '8px'
    };

    switch (pos) {
      case 'top-left':
        return { ...baseStyle, top: '20px', left: '20px' };
      case 'top-right':
        return { ...baseStyle, top: '20px', right: '20px' };
      case 'bottom-left':
        return { ...baseStyle, bottom: '20px', left: '20px' };
      case 'bottom-right':
        return { ...baseStyle, bottom: '20px', right: '20px' };
      case 'top-center':
        return { ...baseStyle, top: '20px', left: '50%', transform: 'translateX(-50%)' };
      case 'bottom-center':
        return { ...baseStyle, bottom: '20px', left: '50%', transform: 'translateX(-50%)' };
      default:
        return { ...baseStyle, top: '20px', right: '20px' };
    }
  };

  return (
    <div>
      <div style={{ marginBottom: '16px' }}>
        <label style={{ display: 'block', marginBottom: '8px', fontWeight: 'bold' }}>
          Toast Position:
        </label>
        <select 
          value={position} 
          onChange={(e) => setPosition(e.target.value)}
          style={{ padding: '8px', borderRadius: '4px', border: '1px solid #d9d9d9' }}
        >
          <option value="top-right">Top Right</option>
          <option value="top-left">Top Left</option>
          <option value="top-center">Top Center</option>
          <option value="bottom-right">Bottom Right</option>
          <option value="bottom-left">Bottom Left</option>
          <option value="bottom-center">Bottom Center</option>
        </select>
      </div>

      <HyButton onClick={() => showToast(`Toast from ${position}`)}>
        Show Toast at {position}
      </HyButton>
      
      <div style={getPositionStyle(position)}>
        {toasts.filter(toast => toast.position === position).map(toast => (
          <div key={toast.id} style={{
            background: '#333',
            color: 'white',
            padding: '12px 16px',
            borderRadius: '6px',
            boxShadow: '0 4px 12px rgba(0, 0, 0, 0.15)',
            minWidth: '200px'
          }}>
            {toast.text}
          </div>
        ))}
      </div>
    </div>
  );
};

## Auto-dismiss and Actions

export const ToastActionsExample = () => {
  const [toasts, setToasts] = useState([]);

  const showToast = (message, options = {}) => {
    const newToast = {
      id: Date.now(),
      text: message,
      ...options
    };
    setToasts(prev => [...prev, newToast]);
    
    if (!options.persistent) {
      setTimeout(() => {
        setToasts(prev => prev.filter(toast => toast.id !== newToast.id));
      }, options.duration || 3000);
    }
  };

  const dismissToast = (toastId) => {
    setToasts(prev => prev.filter(toast => toast.id !== toastId));
  };

  return (
    <div style={{ display: 'flex', gap: '12px', flexWrap: 'wrap' }}>
      <HyButton onClick={() => showToast('Auto-dismiss in 3s', { duration: 3000 })}>
        Auto-dismiss (3s)
      </HyButton>
      <HyButton onClick={() => showToast('Auto-dismiss in 5s', { duration: 5000 })}>
        Auto-dismiss (5s)
      </HyButton>
      <HyButton onClick={() => showToast('Persistent toast', { persistent: true })}>
        Persistent Toast
      </HyButton>
      
      <div style={{
        position: 'fixed',
        top: '20px',
        right: '20px',
        zIndex: 1000,
        display: 'flex',
        flexDirection: 'column',
        gap: '8px'
      }}>
        {toasts.map(toast => (
          <div key={toast.id} style={{
            background: '#333',
            color: 'white',
            padding: '12px 16px',
            borderRadius: '6px',
            boxShadow: '0 4px 12px rgba(0, 0, 0, 0.15)',
            display: 'flex',
            justifyContent: 'space-between',
            alignItems: 'center',
            minWidth: '250px'
          }}>
            <span>{toast.text}</span>
            {toast.persistent && (
              <button
                onClick={() => dismissToast(toast.id)}
                style={{
                  background: 'transparent',
                  border: 'none',
                  color: 'white',
                  cursor: 'pointer',
                  marginLeft: '12px',
                  fontSize: '16px'
                }}
              >
                ✕
              </button>
            )}
          </div>
        ))}
      </div>
    </div>
  );
};

## Progress Indicator

export const ToastProgressExample = () => {
  const [toasts, setToasts] = useState([]);

  const showProgressToast = (message) => {
    const newToast = {
      id: Date.now(),
      text: message,
      progress: 0
    };
    setToasts(prev => [...prev, newToast]);
    
    const interval = setInterval(() => {
      setToasts(prev => 
        prev.map(toast => 
          toast.id === newToast.id 
            ? { ...toast, progress: Math.min(toast.progress + 10, 100) }
            : toast
        )
      );
    }, 300);

    setTimeout(() => {
      clearInterval(interval);
      setToasts(prev => prev.filter(toast => toast.id !== newToast.id));
    }, 3000);
  };

  return (
    <div>
      <HyButton onClick={() => showProgressToast('Processing...')}>
        Show Progress Toast
      </HyButton>
      
      <div style={{
        position: 'fixed',
        top: '20px',
        right: '20px',
        zIndex: 1000,
        display: 'flex',
        flexDirection: 'column',
        gap: '8px'
      }}>
        {toasts.map(toast => (
          <div key={toast.id} style={{
            background: '#333',
            color: 'white',
            borderRadius: '6px',
            boxShadow: '0 4px 12px rgba(0, 0, 0, 0.15)',
            overflow: 'hidden',
            minWidth: '250px'
          }}>
            <div style={{ padding: '12px 16px' }}>
              {toast.text}
            </div>
            <div style={{
              height: '3px',

## API Reference

### Properties

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `toasts` | `ToastItem[]` | `[]` | Array of toast messages to display |
| `position` | `'top-left' \| 'top-right' \| 'top-center' \| 'bottom-left' \| 'bottom-right' \| 'bottom-center'` | `'top-right'` | Position of the toast container |
| `duration` | `number` | `3000` | Default auto-dismiss duration in milliseconds |
| `maxToasts` | `number` | `5` | Maximum number of toasts to show simultaneously |

### Methods

| Method | Parameters | Description |
|--------|------------|-------------|
| `showToast(message, options?)` | `message: string, options?: ToastOptions` | Display a new toast message |
| `dismissToast(id)` | `id: string \| number` | Dismiss a specific toast |
| `clearAll()` | None | Clear all visible toasts |

### Types

#### ToastItem

```typescript
interface ToastItem {
  id: string | number;
  text: string;
  type?: 'success' | 'error' | 'warning' | 'info';
  duration?: number;
  persistent?: boolean;
  action?: {
    label: string;
    handler: () => void;
  };
}
```

#### ToastOptions

```typescript
interface ToastOptions {
  type?: 'success' | 'error' | 'warning' | 'info';
  duration?: number;
  persistent?: boolean;
  action?: {
    label: string;
    handler: () => void;
  };
}
```

### Events

| Event | Description | Event Detail |
|-------|-------------|--------------|
| `toast-show` | Fired when a toast is displayed | `{ toast: ToastItem }` |
| `toast-dismiss` | Fired when a toast is dismissed | `{ toast: ToastItem }` |
| `toast-action` | Fired when a toast action is clicked | `{ toast: ToastItem, action: string }` |

### CSS Custom Properties

You can customize the toast appearance using CSS custom properties:

```css
hy-toast {
  --toast-background: #333333;
  --toast-color: #ffffff;
  --toast-border-radius: 6px;

  --toast-font-size: 14px;

  --toast-max-width: 400px;
  
  /* Type-specific colors */
  --toast-success-background: #52c41a;
  --toast-error-background: #ff4d4f;

```

## Examples

### Network Status Toast

```javascript
// Show connection status
window.addEventListener('online', () => {
  toast.showToast('Connected to internet', { type: 'success' });
});

window.addEventListener('offline', () => {
  toast.showToast('No internet connection', { type: 'error', persistent: true });
});
```

### Form Validation Toasts

```javascript
// Form submission feedback
const submitForm = async (formData) => {
  try {
    toast.showToast('Submitting form...', { type: 'info' });
    
    const result = await api.submit(formData);
    
    toast.showToast('Form submitted successfully!', { type: 'success' });
  } catch (error) {
    toast.showToast('Failed to submit form. Please try again.', { 
      type: 'error',
      action: {
        label: 'Retry',
        handler: () => submitForm(formData)
      }
    });
  }
};
```

### Undo Action Toast

```javascript
const deleteItem = (item) => {
  // Temporarily hide item
  hideItem(item);
  
  toast.showToast('Item deleted', {
    type: 'info',
    duration: 5000,
    action: {
      label: 'Undo',
      handler: () => {
        restoreItem(item);
        toast.showToast('Item restored', { type: 'success' });
      }
    }
  });
};
```
