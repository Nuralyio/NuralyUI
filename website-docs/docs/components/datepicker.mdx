# DatePicker

import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";
import "@nuralyui/datepicker";
import { HyDatePicker } from "@nuralyui/datepicker/react";
import { useState } from "react";

The DatePicker component allows users to select dates from a calendar interface. It supports single date selection, date ranges, different formats, and localization.

## How to use it

<Tabs>
  <TabItem value="npm" label="npm">

```shell
npm install @nuralyui/datepicker
```

</TabItem>
 <TabItem value="yarn" label="yarn">

```shell
yarn add @nuralyui/datepicker
```

</TabItem>
</Tabs>

## Import

<Tabs>
  <TabItem value="html" label="HTML">

```html
<script type="module">
  import '@nuralyui/datepicker';
</script>

<hy-datepicker 
  placeholder="Select a date"
  format="YYYY-MM-DD">
</hy-datepicker>
```

</TabItem>
  <TabItem value="react" label="React">

```jsx
import { HyDatePicker } from '@nuralyui/datepicker/react';

function App() {
  return (
    <HyDatePicker 
      placeholder="Select a date"
      format="YYYY-MM-DD"
      onChange={(date) => console.log('Selected date:', date)}
    />
  );
}
```

## Form Integration

export const DatePickerFormExample = () => {
  const [formData, setFormData] = useState({
    startDate: '',
    endDate: '',
    birthDate: ''
  });

  const handleDateChange = (field, value) => {
    setFormData(prev => ({
      ...prev,
      [field]: value
    }));
  };

  return (
    <form style={{ display: 'flex', flexDirection: 'column', gap: '16px', maxWidth: '400px' }}>
      <div>
        <label style={{ display: 'block', marginBottom: '4px', fontWeight: 'bold' }}>
          Start Date *
        </label>
        <HyDatePicker 
          placeholder="Select start date"
          value={formData.startDate}
          onChange={(value) => handleDateChange('startDate', value)}
          required
        />
      </div>
      
      <div>
        <label style={{ display: 'block', marginBottom: '4px', fontWeight: 'bold' }}>
          End Date *
        </label>
        <HyDatePicker 
          placeholder="Select end date"
          value={formData.endDate}
          onChange={(value) => handleDateChange('endDate', value)}
          required
        />
      </div>
      
      <div>
        <label style={{ display: 'block', marginBottom: '4px', fontWeight: 'bold' }}>
          Birth Date
        </label>
        <HyDatePicker 
          placeholder="Select birth date"
          value={formData.birthDate}
          onChange={(value) => handleDateChange('birthDate', value)}
          format="MM/DD/YYYY"
        />
      </div>
      
      <div style={{ marginTop: '16px', padding: '12px', backgroundColor: '#f5f5f5', borderRadius: '4px' }}>
        <strong>Form Data:</strong>
        <pre style={{ margin: '8px 0 0 0', fontSize: '12px' }}>
          {JSON.stringify(formData, null, 2)}
        </pre>
      </div>
    </form>
  );
};

## API Reference

### Properties

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `placeholder` | `string` | `'Select a date'` | Placeholder text for the input |
| `value` | `string` | `undefined` | Current selected date value |
| `format` | `string` | `'YYYY-MM-DD'` | Date format pattern |
| `mode` | `'day' \| 'month' \| 'year'` | `'day'` | Selection mode |
| `range` | `boolean` | `false` | Enable date range selection |
| `disabled` | `boolean` | `false` | Whether the datepicker is disabled |
| `size` | `'small' \| 'medium' \| 'large'` | `'medium'` | Size of the input |
| `locale` | `string` | `'en'` | Locale for date formatting |
| `minDate` | `string` | `undefined` | Minimum selectable date |
| `maxDate` | `string` | `undefined` | Maximum selectable date |
| `openedCalendar` | `boolean` | `false` | Controls calendar visibility |
| `name` | `string` | `undefined` | Name attribute for form usage |

### Events

| Event | Description | Event Detail |
|-------|-------------|--------------|
| `change` | Fired when date selection changes | `{ value: string \| string[], formattedValue: string }` |
| `open` | Fired when calendar opens | `{}` |
| `close` | Fired when calendar closes | `{}` |


## Format Patterns

The datepicker supports various format patterns based on Day.js:

| Pattern | Description | Example |
|---------|-------------|---------|
| `YYYY` | 4-digit year | 2023 |
| `YY` | 2-digit year | 23 |
| `MM` | Month with leading zero | 01-12 |
| `M` | Month without leading zero | 1-12 |
| `MMMM` | Full month name | January |
| `MMM` | Short month name | Jan |
| `DD` | Day with leading zero | 01-31 |
| `D` | Day without leading zero | 1-31 |
| `dddd` | Full day name | Monday |
| `ddd` | Short day name | Mon |
