# Checkbox

import { HyCheckbox } from "@nuralyui/checkbox/react";
import { useState } from "react";

A versatile checkbox component that allows users to select one or more options from a set. The checkbox supports multiple states including checked, unchecked, indeterminate, and disabled states, with customizable sizes for different use cases.

## Installation

```bash
npm install @nuralyui/checkbox
```

## Usage

### React

```jsx
import { HyCheckbox } from "@nuralyui/checkbox/react";
function App() {
  const [checked, setChecked] = useState(false);
  const handleChange = (event) => {
    setChecked(event.detail.value);
  };
  return (
    <HyCheckbox 
      checked={checked} 
      onCheckboxChanged={handleChange}
    >
      Accept terms and conditions
    </HyCheckbox>
  );
}
```

export const CheckboxExample = ({ 
  checked = false, 
  disabled = false, 
  indeterminate = false, 
  size = "medium",
  children = "Checkbox Label"
}) => {
  const [isChecked, setIsChecked] = useState(checked);
  const [isIndeterminate, setIsIndeterminate] = useState(indeterminate);
  const handleChange = (event) => {
    setIsChecked(event.detail.value);
    if (isIndeterminate) {
      setIsIndeterminate(false);
    }
  };
  return (
    <HyCheckbox
      checked={isChecked}
      disabled={disabled}
      indeterminate={isIndeterminate}
      size={size}
      onCheckboxChanged={handleChange}
    >
      {children}
    </HyCheckbox>
  );
};

export const CheckboxGroupExample = () => {
  const [parentChecked, setParentChecked] = useState(false);
  const [parentIndeterminate, setParentIndeterminate] = useState(false);
  const [childChecked, setChildChecked] = useState([false, false, false]);
  const handleParentChange = (event) => {
    const newValue = event.detail.value;
    setParentChecked(newValue);
    setParentIndeterminate(false);
    setChildChecked([newValue, newValue, newValue]);
  };
  const handleChildChange = (index) => (event) => {
    const newChildChecked = [...childChecked];
    newChildChecked[index] = event.detail.value;
    setChildChecked(newChildChecked);
    const checkedCount = newChildChecked.filter(Boolean).length;
    setParentChecked(checkedCount === newChildChecked.length);
    setParentIndeterminate(checkedCount > 0 && checkedCount < newChildChecked.length);
  };
  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: '8px' }}>
      <HyCheckbox
        checked={parentChecked}
        indeterminate={parentIndeterminate}
        onCheckboxChanged={handleParentChange}
      >
        Select All Options
      </HyCheckbox>
      <div style={{ marginLeft: '24px', display: 'flex', flexDirection: 'column', gap: '4px' }}>
        <HyCheckbox checked={childChecked[0]} onCheckboxChanged={handleChildChange(0)}>
          Option 1
        </HyCheckbox>
        <HyCheckbox checked={childChecked[1]} onCheckboxChanged={handleChildChange(1)}>
          Option 2
        </HyCheckbox>
        <HyCheckbox checked={childChecked[2]} onCheckboxChanged={handleChildChange(2)}>
          Option 3
        </HyCheckbox>
      </div>
    </div>
  );
};

## When To Use

- **Form Inputs**: For collecting user preferences, settings, or consent
- **Multi-Selection**: When users need to select multiple options from a list
- **Toggle Settings**: For enabling/disabling features or preferences
- **Agreements**: For terms of service, privacy policies, or consent forms
- **Filtering**: For selecting filter criteria in search or data display interfaces
- **Bulk Actions**: For selecting multiple items to perform batch operations

## Props/Attributes

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `checked` | `boolean` | `false` | Whether the checkbox is checked |
| `disabled` | `boolean` | `false` | Whether the checkbox is disabled |
| `indeterminate` | `boolean` | `false` | Whether the checkbox is in indeterminate state |
| `size` | `'small' \| 'medium' \| 'large'` | `'medium'` | Size variant of the checkbox |

### Size Variants

- **`small`**: Compact checkbox for dense layouts
- **`medium`**: Standard checkbox size (default)
- **`large`**: Larger checkbox for better accessibility and prominence

### States

- **`checked`**: The checkbox is selected/checked
- **`unchecked`**: The checkbox is not selected (default)
- **`indeterminate`**: The checkbox is in a partially selected state (useful for parent-child relationships)
- **`disabled`**: The checkbox cannot be interacted with

## Events

| Event | Description | Detail |
|-------|-------------|--------|
| `checkbox-changed` | Fired when checkbox state changes | `{ value: boolean }` |

## Advanced Usage

### Controlled Checkbox

```jsx
function ControlledCheckbox() {
  const [isChecked, setIsChecked] = useState(false);
  const handleChange = (event) => {
    setIsChecked(event.detail.value);
    // Perform additional actions
    if (event.detail.value) {
      console.log('Checkbox was checked');
    }
  };
  return (
    <HyCheckbox checked={isChecked} onCheckboxChanged={handleChange}>
      Controlled checkbox
    </HyCheckbox>
  );
}
```

### Checkbox Group Management

```jsx
function CheckboxGroup() {
  const [selectedItems, setSelectedItems] = useState([]);
  const options = ['Option 1', 'Option 2', 'Option 3'];
  const handleItemChange = (item) => (event) => {
    if (event.detail.value) {
      setSelectedItems([...selectedItems, item]);
    } else {
      setSelectedItems(selectedItems.filter(i => i !== item));
    }
  };
  return (
    <div>
      {options.map(option => (
        <HyCheckbox
          key={option}
          checked={selectedItems.includes(option)}
          onCheckboxChanged={handleItemChange(option)}
        >
          {option}
        </HyCheckbox>
      ))}
    </div>
  );
}
```

### Form Integration

```jsx
function FormWithCheckboxes() {
  const [formData, setFormData] = useState({
    notifications: false,
    newsletter: false,
    terms: false
  });
  const handleCheckboxChange = (field) => (event) => {
    setFormData({
      ...formData,
      [field]: event.detail.value
    });
  };
  const handleSubmit = (e) => {
    e.preventDefault();
    console.log('Form data:', formData);
  };
  return (
    <form onSubmit={handleSubmit}>
      <HyCheckbox
        checked={formData.notifications}
        onCheckboxChanged={handleCheckboxChange('notifications')}
      >
        Enable notifications
      </HyCheckbox>
      <HyCheckbox
        checked={formData.newsletter}
        onCheckboxChanged={handleCheckboxChange('newsletter')}
      >
        Subscribe to newsletter
      </HyCheckbox>
      <HyCheckbox
        checked={formData.terms}
        onCheckboxChanged={handleCheckboxChange('terms')}
      >
        I agree to the terms and conditions
      </HyCheckbox>
      <button type="submit" disabled={!formData.terms}>
        Submit
      </button>
    </form>
  );
}
```

## Accessibility

The checkbox component includes comprehensive accessibility features:

- **Keyboard Navigation**: Use Tab to focus and Space to toggle
- **ARIA Attributes**: Proper ARIA roles and states for screen readers
- **Focus Management**: Clear visual focus indicators
- **Semantic HTML**: Uses native checkbox input elements
- **Screen Reader Support**: Descriptive labels and state announcements
- **High Contrast**: Works well with high contrast modes

### Keyboard Interactions

| Key | Action |
|-----|--------|
| `Tab` | Move focus to/from the checkbox |
| `Space` | Toggle checkbox state |
| `Enter` | Toggle checkbox state (when focused) |

## Best Practices

1. **Clear Labels**: Use descriptive labels that clearly explain what the checkbox controls
2. **Logical Grouping**: Group related checkboxes together visually and semantically
3. **Default States**: Set sensible defaults based on user expectations
4. **Required Indicators**: Clearly mark required checkboxes in forms
5. **Error States**: Provide clear feedback for validation errors
6. **Consistent Sizing**: Use consistent checkbox sizes within the same interface
7. **Indeterminate Usage**: Use indeterminate state only for parent-child relationships
8. **Accessibility**: Always provide accessible labels and proper keyboard navigation

## Common Patterns

### Settings Panel

```jsx
function SettingsPanel() {
  const [settings, setSettings] = useState({
    darkMode: false,
    autoSave: true,
    notifications: true,
    beta: false
  });
  const updateSetting = (key) => (event) => {
    setSettings({
      ...settings,
      [key]: event.detail.value

  };
  return (
    <div className="settings-panel">
      <h3>Preferences</h3>
      <HyCheckbox checked={settings.darkMode} onCheckboxChanged={updateSetting('darkMode')}>
        Enable dark mode
      </HyCheckbox>
      <HyCheckbox checked={settings.autoSave} onCheckboxChanged={updateSetting('autoSave')}>
        Auto-save changes
      </HyCheckbox>
      <HyCheckbox checked={settings.notifications} onCheckboxChanged={updateSetting('notifications')}>
        Push notifications
      </HyCheckbox>
      <HyCheckbox checked={settings.beta} onCheckboxChanged={updateSetting('beta')}>
        Enable beta features
      </HyCheckbox>
    </div>
  );
}
```

### Terms and Conditions

```jsx
function TermsAgreement() {
  const [agreements, setAgreements] = useState({
    terms: false,
    privacy: false,
    marketing: false
  });
  const isValid = agreements.terms && agreements.privacy;
  const handleChange = (field) => (event) => {
    setAgreements({
      ...agreements,
      [field]: event.detail.value
    });
  };
  return (
    <div className="terms-section">
      <HyCheckbox checked={agreements.terms} onCheckboxChanged={handleChange('terms')}>
        I agree to the <a href="/terms">Terms of Service</a> *
      </HyCheckbox>
      <HyCheckbox checked={agreements.privacy} onCheckboxChanged={handleChange('privacy')}>
        I agree to the <a href="/privacy">Privacy Policy</a> *
      </HyCheckbox>
      <HyCheckbox checked={agreements.marketing} onCheckboxChanged={handleChange('marketing')}>
        I would like to receive marketing communications
      </HyCheckbox>
      <button disabled={!isValid}>Continue</button>
    </div>
  );
}
```
