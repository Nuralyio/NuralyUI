# Tabs

import { HyTabsComponent } from "@nuralyui/tabs/react";
import { useState } from "react";
import { html } from "lit";
import "@nuralyui/icon";
import { HyButton } from "@nuralyui/button/react";

A comprehensive tab component that provides navigation between different views and content sections. The tabs component supports various orientations, alignment options, and interactive features like drag-and-drop reordering, inline editing, and dynamic tab management.

## Installation

```bash
npm install @nuralyui/tabs
```

## Usage

### React

```jsx
import { HyTabsComponent } from "@nuralyui/tabs/react";

function App() {
  const [tabs] = useState([
    { label: "Tab 1", content: "Content for Tab 1" },
    { label: "Tab 2", content: "Content for Tab 2" },
    { label: "Tab 3", content: "Content for Tab 3" }
  ]);

  return (
    <HyTabsComponent tabs={tabs} activeTab={0} />
  );
}
```

### Web Components

```html
<hy-tabs active-tab="0">
  <div label="Tab 1">Content for Tab 1</div>
  <div label="Tab 2">Content for Tab 2</div>
  <div label="Tab 3">Content for Tab 3</div>
</hy-tabs>
```

export const swapElements = (arr, index1, index2) => {
  arr = arr;
  const temp = arr[index1];
  arr[index1] = arr[index2];
  arr[index2] = temp;
  return arr;
};

export const SimpleTabExample = ({
  children,
  orientation = "horizontal",
  tabsAlign = "",
  editable = {
    canDeleteTab: false,
    canEditTabTitle: false,
    canAddTab: false,
    canMove: false,
  },
}) => {
  const [editableTabs, setEditableTabs] = useState([
    {
      label: "Tab 1",
      content: `Content for Tab 1`,
    },
    {
      label: "Tab 2",
      content: `Content for Tab 2`,
    },
    {
      label: "Tab 3",
      content: <strong> bold content for Tab 3</strong>,
    },
    {
      label: "Tab 4",
      content: `Content for Tab 4`,
    },
  ]);
  const [activeTab, setActiveTab] = useState(0);
  return (
    <>
      <HyTabsComponent
        activeTab={activeTab}
        tabs={editableTabs}
        tabOrderChange={function (e) {
          setEditableTabs([
            ...swapElements(
              editableTabs,
              e.detail.sourceIndex,
              e.detail.targetIndex
            ),
          ]);
          setActiveTab(e.detail.targetIndex);
        }}
        removeTab={function (e) {
          editableTabs.splice(e.detail.index, 1);
          setEditableTabs([...editableTabs]);
        }}
        editable={editable}
        tabsAlign={tabsAlign}
        orientation={orientation}
        addTab={function (e) {
          console.log(e);
          setEditableTabs([
            ...editableTabs,
            {
              label: "Tab " + (editableTabs.length + 1),
              content: `Content for Tab ${editableTabs.length + 1}`,
            },
          ]);
        }}
      />
    </>
  );
};

## When To Use

- **Content Organization**: When you need to organize related content into logical sections
- **Space Efficiency**: When you want to display multiple views in a limited space
- **Navigation**: When users need to switch between different but related information
- **Dashboard Layouts**: For organizing different data views or functional areas
- **Settings Panels**: For grouping configuration options into categories

## Props/Attributes

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `activeTab` | `number` | `0` | Index of the currently active tab |
| `orientation` | `'horizontal' \| 'vertical'` | `'horizontal'` | Layout orientation of the tabs |
| `tabsAlign` | `'left' \| 'center' \| 'right'` | `'left'` | Alignment of the tab headers |
| `editable` | `TabEditable` | `{}` | Configuration for interactive tab features |
| `tabs` | `TabItem[]` | `[]` | Array of tab data objects |

### TabEditable Interface

```typescript
interface TabEditable {
  canDeleteTab?: boolean;    // Allow removing tabs
  canEditTabTitle?: boolean; // Allow inline editing of tab titles
  canAddTab?: boolean;       // Show add tab button
  canMove?: boolean;         // Allow drag-and-drop reordering
}
```

### TabItem Interface

```typescript
interface TabItem {
  label: string;           // Tab title
  content: React.ReactNode; // Tab content
  editable?: TabEditable;  // Per-tab editable settings (overrides global)
}
```

## Events

| Event | Description | Detail |
|-------|-------------|--------|
| `removeTab` | Fired when a tab is removed | `{ index: number }` |
| `tabEdited` | Fired when a tab title is edited | `{ tab: { label: string, index: number } }` |
| `tabTilteClick` | Fired when a tab header is clicked | `{ index: number }` |
| `tabOrderChange` | Fired when tabs are reordered | `{ sourceIndex: number, targetIndex: number, sourceTab: Element, targetTab: Element }` |
| `addTab` | Fired when the add tab button is clicked | `{}` |

### Interactive Complete Example

Explore all features combined in this interactive demo:

export const InteractiveTabExample = ({}) => {
  const [editing, setEditing] = useState(false);
  const [closing, setClosing] = useState(false);
  const [adding, setAdding] = useState(false);
  const [moving, setMoving] = useState(false);
  const [tabsAlign, setTabsAlign] = useState("left");
  const [orientation, setOrientation] = useState("horizontal");
  return (
    <>
      <div style={{ marginBottom: '1rem', display: 'flex', gap: '0.5rem', flexWrap: 'wrap' }}>
        <HyButton
          type={editing ? "primary" : ""}
          onClick={() => {
            setEditing(!editing);
          }}
        >
          {editing ? "Disable" : "Enable"} editing
        </HyButton>
        <HyButton
          type={closing ? "primary" : ""}
          onClick={() => {
            setClosing(!closing);
          }}
        >
          {closing ? "Disable" : "Enable"} closing
        </HyButton>
        <HyButton
          type={adding ? "primary" : ""}
          onClick={() => {
            setAdding(!adding);
          }}
        >
          {adding ? "Disable" : "Enable"} Adding
        </HyButton>
        <HyButton
          type={moving ? "primary" : ""}
          onClick={() => {
            setMoving(!moving);
          }}
        >
          {moving ? "Disable" : "Enable"} Moving
        </HyButton>
        <HyButton
          onClick={() => {
            setTabsAlign(tabsAlign == "left" ? "center" : tabsAlign == "center" ? "right" : "left");
          }}
        >
          Align: {tabsAlign}
        </HyButton>
        <HyButton
          onClick={() => {
            setOrientation(
              orientation == "horizontal" ? "vertical" : "horizontal"
            );
          }}
        >
          {orientation}
        </HyButton>
      </div>
      <SimpleTabExample
        tabsAlign={tabsAlign}
        orientation={orientation}
        editable={{
          canDeleteTab: closing,
          canEditTabTitle: editing,
          canAddTab: adding,
          canMove: moving,
        }}
      />
    </>
  );
};

## Accessibility

The tabs component includes built-in accessibility features:

- **Keyboard Navigation**: Use arrow keys to navigate between tabs and Enter/Space to activate
- **ARIA Attributes**: Proper ARIA roles and labels for screen readers
- **Focus Management**: Clear visual focus indicators and logical tab order
- **Semantic HTML**: Uses appropriate HTML elements for better accessibility

## Best Practices

1. **Limit Tab Count**: Keep the number of tabs reasonable (typically 3-7) to avoid overwhelming users
2. **Clear Labels**: Use concise, descriptive tab labels that clearly indicate the content
3. **Consistent Content**: Ensure related content is grouped logically within tabs
4. **Responsive Design**: Consider how tabs will behave on smaller screens
5. **Loading States**: Provide feedback when tab content is loading asynchronously
