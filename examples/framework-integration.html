📦 Nuraly UI - Module Integration Guide

Learn how to integrate Nuraly UI with modern build tools and frameworks

🎯 Framework Integration Examples

Vanilla JavaScript / HTML

// main.js import '@nuralyui/button'; import '@nuralyui/input'; import '@nuralyui/modal'; // Your application code document.querySelector('#open-modal-btn') .addEventListener('click', () => { const modal = document.querySelector('nr-modal'); modal.show(); });
<!-- index.html --> <!DOCTYPE html> <html lang="en"> <head> <script type="module" src="/src/main.js"></script> </head> <body> <nr-button id="open-modal-btn" variant="primary">Open Modal</nr-button> <nr-modal id="my-modal" title="Hello"> <p>This is a modal!</p> </nr-modal> </body> </html>

React Integration

Note: Nuraly UI provides React wrappers for all components!
// App.jsx import { useState } from 'react'; import { NrButton } from '@nuralyui/button/react'; import { NrInput } from '@nuralyui/input/react'; import { NrModal } from '@nuralyui/modal/react'; function App() { const [modalOpen, setModalOpen] = useState(false); const [inputValue, setInputValue] = useState(''); return ( <div> <NrInput label="Name" value={inputValue} onInput={(e) => setInputValue(e.target.value)} /> <NrButton variant="primary" onClick={() => setModalOpen(true)} > Open Modal </NrButton> <NrModal open={modalOpen} onClose={() => setModalOpen(false)} title="Hello" > <p>Input value: {inputValue}</p> </NrModal> </div> ); } export default App;
// Or use web components directly import '@nuralyui/button'; function App() { return ( <nr-button variant="primary" onClick={(e) => console.log('clicked')} > Click me </nr-button> ); }

Vue 3 Integration

// main.js import { createApp } from 'vue'; import App from './App.vue'; // Import components import '@nuralyui/button'; import '@nuralyui/input'; import '@nuralyui/modal'; createApp(App).mount('#app');
<!-- App.vue --> <template> <div> <nr-input :label="'Name'" :value="inputValue" @input="inputValue = $event.target.value" ></nr-input> <nr-button variant="primary" @click="modalOpen = true" > Open Modal </nr-button> <nr-modal :open="modalOpen" @close="modalOpen = false" title="Hello" > <p>Input value: {{ inputValue }}</p> </nr-modal> </div> </template> <script setup> import { ref } from 'vue'; const modalOpen = ref(false); const inputValue = ref(''); </script>
Vue Configuration: Add to vite.config.js or vue.config.js to properly handle custom elements:
// vite.config.js export default { plugins: [ vue({ template: { compilerOptions: { isCustomElement: (tag) => tag.startsWith('nr-') } } }) ] }

Svelte Integration

<!-- App.svelte --> <script> import { onMount } from 'svelte'; let inputValue = ''; let modalOpen = false; let modalElement; onMount(async () => { // Dynamic import for better code splitting await import('@nuralyui/button'); await import('@nuralyui/input'); await import('@nuralyui/modal'); }); function openModal() { modalElement?.show(); } </script> <div> <nr-input label="Name" bind:value={inputValue} on:input={(e) => inputValue = e.target.value} ></nr-input> <nr-button variant="primary" on:click={openModal} > Open Modal </nr-button> <nr-modal bind:this={modalElement} title="Hello" > <p>Input value: {inputValue}</p> </nr-modal> </div>

Angular Integration

// app.module.ts import { NgModule, CUSTOM_ELEMENTS_SCHEMA } from '@angular/core'; import { BrowserModule } from '@angular/platform-browser'; import { AppComponent } from './app.component'; // Import web components import '@nuralyui/button'; import '@nuralyui/input'; import '@nuralyui/modal'; @NgModule({ declarations: [AppComponent], imports: [BrowserModule], schemas: [CUSTOM_ELEMENTS_SCHEMA], // Important! bootstrap: [AppComponent] }) export class AppModule { }
// app.component.ts import { Component } from '@angular/core'; @Component({ selector: 'app-root', template: ` <nr-input [attr.label]="'Name'" [attr.value]="inputValue" (input)="onInput($event)" ></nr-input> <nr-button variant="primary" (click)="openModal()" > Open Modal </nr-button> <nr-modal #modal [attr.title]="'Hello'" > <p>Input value: {{ inputValue }}</p> </nr-modal> ` }) export class AppComponent { inputValue = ''; onInput(event: any) { this.inputValue = event.target.value; } openModal() { const modal = document.querySelector('nr-modal') as any; modal?.show(); } }

🎨 Styling & Theming

Using CSS Custom Properties

/* styles.css */ :root { /* Primary colors */ --nr-primary-color: #667eea; --nr-secondary-color: #764ba2; /* Typography */ --nr-font-family: 'Inter', -apple-system, sans-serif; --nr-font-size-base: 16px; /* Spacing */ --nr-spacing-sm: 8px; --nr-spacing-md: 16px; --nr-spacing-lg: 24px; /* Border radius */ --nr-border-radius: 6px; }

Using Pre-built Themes

// Import theme CSS import '@nuralyui/themes/carbon.css'; // or import '@nuralyui/themes/default.css';

📦 Package Bundle Usage

Tip: Use package bundles to import multiple related components at once!
// Import all form components import '@nuralyui/forms'; // Includes: input, textarea, checkbox, radio, select, datepicker, etc. // Import all common components import '@nuralyui/common'; // Includes: icon, badge, divider, label // Import all layout components import '@nuralyui/layout'; // Includes: layout, grid, flex, card

React Bundle Usage
